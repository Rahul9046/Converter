 25 /**
 26  * A method that checks if a particular charcter is present inside a string.
 27  * @function
 28  * @param {string} toCheck-The character to be checked inside a string
 29  * @returns {boolean} 
 30  * @class 
 31  */
 32 String.prototype.contains = function (toCheck) {
 33   var splitStr = this.split('');
 34   for (var i = 0; i < splitStr.length; i++) {
 35     if (splitStr[i] === toCheck) {
 36       return true;
 37     }
 38   }
 39   return false;
 40 };
 41 /**
 42  * The decider class
 43  * @class
 44  */
 45 class Decider {
 46   /**
 47    * @constructor
 48    * @param {array} arr-The array that contains alpha-numeric words which will be converted.
 49    */
 50   constructor (arr) {
 51     this.deciderArray = arr;
 52   }
 53   /**
 54    * The callback which will be called after each conversion
 55    * @callback
 56    * @param {number} index-The index of the word in the main string which was passes as value in the text box. 
 57    * @param {number} value-The output which will be updated in the main string. 
 58    * @param {number} mainArr-The array that contains the outputs along with the index of each in the main string.  
 59    */
 60   callbackToUpdate (index, value, mainArr) {
 61     mainArr[index] = value;
 62   }
 63   /**
 64    * The method that decides what type of alphanumeric string each input is and then sends it to its respective module for conversion.
 65    * @function 
 66    * @param {arr} arr-The array that contains the alphanumeric texts
 67    */
 68   decider (arr) {
 69     var deciderArr;
 70     for (var i = 0; i < this.deciderArray.length; i++) {
 71       if (isNaN(this.deciderArray[i][0]) === false && (this.deciderArray[i][0].contains('.') === false)) {
 72         deciderArr = this.deciderArray;
 73         (function (j, deciderArray, that, arr) {
 74           var parsing = setTimeout(function (callbackToUpdate) {
 75             const cardinalObj = new Cardinal(deciderArray[j][0]);
 76             deciderArray[j][2] = 1;
 77             var convertedValue = cardinalObj.converter();
 78             that.callbackToUpdate(deciderArray[j][1], convertedValue, parsing, arr);
 79           }, 0);
 80         })(i, deciderArr, this, arr);
 81       } else if (this.deciderArray[i][0].contains(':') || this.deciderArray[i][0].contains('.') || this.deciderArray[i][0].contains('-') || this.deciderArray[i][0].contains('+') || this.deciderArray[i][0].contains('/')) {
 82         if (this.deciderArray[i][0].contains(':')) {
 83           deciderArr = this.deciderArray;
 84           (function (j, deciderArray, that, arr) {
 85             var parsing = setTimeout(function (callbackToUpdate) {
 86               const timeObj = new Time(deciderArray[j][0]);
 87               deciderArray[j][2] = 1;
 88               var convertedValue = timeObj.convertTime();
 89               that.callbackToUpdate(deciderArray[j][1], convertedValue, parsing, arr);
 90             }, 0);
 91           })(i, deciderArr, this, arr);
 92         } else if ((this.deciderArray[i][0].contains('-') && this.deciderArray[i][0].split('-').length === 3) || (this.deciderArray[i][0].contains('/') && this.deciderArray[i][0].split('/').length === 3)) {
 93           deciderArr = this.deciderArray;
 94           (function (j, deciderArray, that, arr) {
 95             var parsing = setTimeout(function (callbackToUpdate) {
 96               const dateobj = new Date(deciderArray[j][0]);
 97               deciderArray[j][2] = 1;
 98               var convertedValue = dateobj.convert();
 99               that.callbackToUpdate(deciderArray[j][1], convertedValue, parsing, arr);
100             }, 0);
101           })(i, deciderArr, this, arr);
102         } else if (this.deciderArray[i][0].contains('+') || (this.deciderArray[i][0].contains('-') && this.deciderArray[i][0].split('-').length === 2 && (this.deciderArray[i][0].split('-')[0].contains('+') || this.deciderArray[i][0].split('-')[0] === '0'))) {
103           deciderArr = this.deciderArray;
104           (function (j, deciderArray, that, arr) {
105             var parsing = setTimeout(function (callbackToUpdate) {
106               const phnNumObj = new PhoneNumber(deciderArray[j][0]);
107               deciderArray[j][2] = 1;
108               var convertedValue = phnNumObj.converter();
109               that.callbackToUpdate(deciderArray[j][1], convertedValue, parsing, arr);
110             }, 0);
111           })(i, deciderArr, this, arr);
112         } else if (this.deciderArray[i][0].contains('.') || (this.deciderArray[i][0].contains('-') && this.deciderArray[i][0].split('-').length === 2) || (this.deciderArray[i][0].contains('/') && this.deciderArray[i][0].split('/').length === 2)) {
113           deciderArr = this.deciderArray;
114           (function (j, deciderArray, that, arr) {
115             var parsing = setTimeout(function (callbackToUpdate) {
116               const decFracObj = new DecimalFraction(deciderArray[j][0]);
117               deciderArray[j][2] = 1;
118               var convertedValue = decFracObj.convert();
119               that.callbackToUpdate(deciderArray[j][1], convertedValue, parsing, arr);
120             }, 0);
121           })(i, deciderArr, this, arr);
122         }
123       } else {
124         deciderArr = this.deciderArray;
125         (function (j, deciderArray, that, arr) {
126           var parsing = setTimeout(function (callbackToUpdate) {
127             const sufpreObj = new SuffixPrefix(deciderArray[j][0]);
128             deciderArray[j][2] = 1;
129             var convertedValue = sufpreObj.convert();
130             that.callbackToUpdate(deciderArray[j][1], convertedValue, arr);
131           }, 0);
132         })(i, deciderArr, this, arr);
133       }
134     }
135   }
136 }
137 export default Decider;
