1 /**
  2  * @borrows n2w as n2w
  3  */
  4 import n2w from './n2w.js';
  5 /**
  6  * A method that checks if a separator encountered is in the list of the valid separators.
  7  * @function
  8  * @param {string} toCheck-The separator to be checked inside the array
  9  * @returns {boolean} 
 10  * @class 
 11  */
 12 Array.prototype.contains = function (toCheck) {
 13   for (var i = 0; i < this.length; i++) {
 14     if (this[i] === toCheck) {
 15       return true;
 16     }
 17   }
 18   return false;
 19 };
 20 /**
 21  * A method that checks if the time representation is valid.
 22  * @function
 23  * @returns {boolean} 
 24  * @class 
 25  */
 26 String.prototype.validTime = function () {
 27   var separators = [',', ':'];
 28   var splittedStr = this.split(''), count = 0, flag = 0, lastSeparator, separator;
 29   for (var i = 0; i < splittedStr.length; i++) {
 30     if (separators.contains(splittedStr[i])) {
 31       separator = splittedStr[i];
 32       if (lastSeparator === undefined || lastSeparator === splittedStr[i]) {
 33         flag = 1;
 34         lastSeparator = splittedStr[i];
 35         count++;
 36       } else {
 37         flag = 0;
 38         break;
 39       }
 40     }
 41   }
 42   if (flag) {
 43     return {
 44       'isValid': true,
 45       'count': count,
 46       'separator': separator
 47     };
 48   } else {
 49     return {
 50       'isValid': false
 51     };
 52   }
 53 };
 54 /**
 55  * The Time class
 56  * @class
 57  */
 58 class Time {
 59   /**
 60    * @constructor
 61    * @param {string} str-The time in numeric  representation which will be converted.  
 62    */
 63   constructor (str) {
 64     this.time = str;
 65   }
 66   /**
 67    * The method that converts the time to its textual representation
 68    * @function
 69    * @returns {string}
 70    */
 71   convertTime () {
 72     if (this.time.validTime().isValid) {
 73       var splittedTime = this.time.split(this.time.validTime().separator), parsedTime = '', count = this.time.validTime().count;
 74       var timeAppends = {
 75         '2': 'hour',
 76         '1': 'minute',
 77         '0': 'second'
 78       };
 79       if (count === 2) {
 80         for (var i = 0; i < splittedTime.length; i++) {
 81           if (splittedTime[i] === '00') {
 82             count--;
 83             continue;
 84           } else {
 85             parsedTime += n2w(splittedTime[i]) + ' ' + timeAppends[count] + this.time.validTime().separator;
 86             count--;
 87           }
 88         }
 89       } else {
 90         for (var i = 0; i < splittedTime.length; i++) {
 91           if (splittedTime[i] === '00') {
 92             continue;
 93           } else {
 94             parsedTime += n2w(splittedTime[i]) + this.time.validTime().separator;
 95           }
 96         }
 97       }
 98       return parsedTime.slice(0, parsedTime.length - 1);
 99     } else {
100       return this.time;
101     }
102   }
103 }
104 export default Time;
