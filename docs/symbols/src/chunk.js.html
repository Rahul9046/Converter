1 /**
  2  * @borrows Identifier as Identifier
  3  */
  4 import Identifier from './Identifier.js'
  5 /**
  6  * Takes the input in string,splits it and sends it to Identifier for parsing.
  7  * @function 
  8  * @param {string} str-The input to be parsed 
  9  */
 10 let chunk=(str)=>{
 11         let mainArr =str.split(" ");
 12         var identifierObj = new Identifier(str,mainArr);
 13         identifierObj.getParsed();
 14 }
 15 /**
 16  * Gets the text from the text box and sends it to chunk for parsing.
 17  * @fires document#click
 18  */
 19 document.getElementById("convert").addEventListener("click",
 20 /**
 21  * @event document#click
 22  */
 23 function(){
 24     var text=document.getElementById("input").value;
 25     chunk(text);
 26 });
