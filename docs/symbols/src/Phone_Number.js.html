1 /**
  2  * @borrows n2w as n2w
  3  */
  4 import n2w from './n2w.js';
  5 /**
  6  * The PhoneNumber class
  7  * @class
  8  */
  9 class PhoneNumber {
 10   /**
 11    * @constructor
 12    * @param {string} num-The phone number in numeric format 
 13    */
 14   constructor (num) {
 15     this.num = num;
 16   }
 17   /**
 18    * The method that converts phone numbers to its textual representation.
 19    * @function
 20    * @returns {string}
 21    */
 22   converter () {
 23     var splittedNum = this.num.split(''), parsedNum = '';
 24     for (var i = 0; i < splittedNum.length; i++) {
 25       if (isNaN(splittedNum[i]) === false) {
 26         parsedNum += n2w(splittedNum[i]) + ' ';
 27       } else {
 28         parsedNum += splittedNum[i] + ' ';
 29       }
 30     }
 31     parsedNum = parsedNum.slice(0, parsedNum.length - 1);
 32     return parsedNum;
 33   }
 34 }
 35 export default PhoneNumber;
