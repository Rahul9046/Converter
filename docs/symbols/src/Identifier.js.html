1 /**
  2  * @borrows Decider as Decider
  3  */
  4 import Decider from './Decider.js';
  5 /**
  6  * The Identifier class
  7  * @class
  8  */
  9 class Identifier {
 10   /**
 11    * @constructor
 12    * @param {string} str-The main string which was passed as input to the text box. 
 13    * @param {array} mainArr-The array that contains the splitted texts. 
 14    */
 15   constructor (str, mainArr) {
 16     this.toParse = str;
 17     this.mainArr = mainArr;
 18   }
 19   /**
 20    * The method which filters out the alphanumeric texts and passes it to Decider for further processing.
 21    * @function
 22    */
 23   getParsed () {
 24     const pattern = /[0-9]/;
 25     var splittedText = this.toParse.split(' '), index = 0;
 26     const parsedArray = [];
 27     for (var i = 0; i < splittedText.length; i++) {
 28       if (pattern.test(splittedText[i])) {
 29         parsedArray[index++] = [splittedText[i], i, 0];
 30       }
 31     }
 32     /**
 33      * The event that fires when all the alphanumeric texts have been converted to its string representations.
 34      * @event
 35      * @param {array} mainArr-The array that contains the splitted texts.
 36      */
 37     (function (mainArr) {
 38       var checkForUsedBits = setInterval(function () {
 39         var flag = 1;
 40         for (var i = 0; i < parsedArray.length; i++) {
 41           flag = flag && parsedArray[i][2];
 42         }
 43         if (flag) {
 44           document.getElementById('output').innerHTML = mainArr.join(' ').toString();
 45           clearTimeout(checkForUsedBits);
 46         }
 47       }, 1);
 48     })(this.mainArr);
 49     var deciderObj = new Decider(parsedArray);
 50     deciderObj.decider(this.mainArr);
 51   }
 52 }
 53 export default Identifier;
