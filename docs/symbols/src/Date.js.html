 47 /**
 48  * A method that checks if a particular charcter is present inside a string.
 49  * @function
 50  * @param {string} toCheck-The character to be checked inside a string
 51  * @returns {boolean} 
 52  * @class 
 53  */
 54 String.prototype.contains = function (toCheck) {
 55   var splitStr = this.split('');
 56   for (var i = 0; i < splitStr.length; i++) {
 57     if (splitStr[i] === toCheck) {
 58       return true;
 59     }
 60   }
 61   return false;
 62 };
 63 /**
 64  * The Date class
 65  * @class
 66  */
 67 class Date {
 68   /**
 69    * @constructor
 70    * @param {string} str-The string that represents the date 
 71    */
 72   constructor (str) {
 73     this.date = str;
 74   }
 75   /**
 76    * The method that converts dates to its string representation.
 77    * @function
 78    * @returns {string}
 79    */
 80   convert () {
 81     var parsedStr = '', tobeParsed = '';
 82     if (this.date.contains('/')) {
 83       var splitStr = this.date.split('/');
 84       if (splitStr[0].slice(0, 1) === '0') {
 85         splitStr[0] = splitStr[0].slice(1);
 86       }
 87       if (splitStr[0].length === 2) {
 88         if (splitStr[0].slice(0, 1) === '1') {
 89           tobeParsed = splitStr[0];
 90         } else {
 91           parsedStr += n2w(splitStr[0].slice(0, 1) + '0') + ' ';
 92           tobeParsed = splitStr[0].slice(-1);
 93         }
 94       } else {
 95         tobeParsed += splitStr[0];
 96       }
 97       for (var keys in ordinals) {
 98         if (ordinals[keys][0] === Number(tobeParsed)) {
 99           parsedStr += keys + ' ';
100           tobeParsed = '';
101         }
102       }
103 
104       if (splitStr[1].slice(0, 1) === '0') {
105         splitStr[1] = splitStr[1].slice(1);
106       }
107       for (var keys in date) {
108         if (keys === splitStr[1]) {
109           parsedStr += date[keys] + ' ';
110         }
111       }
112       parsedStr += n2w(splitStr[2]);
113     } else if (this.date.contains('-')) {
114       var splitStr = this.date.split('-');
115       if (splitStr[0].slice(0, 1) === '0') {
116         splitStr[0] = splitStr[0].slice(1);
117       }
118       if (splitStr[0].length === 2) {
119         if (splitStr[0].slice(0, 1) === '1') {
120           tobeParsed = splitStr[0];
121         } else {
122           parsedStr += n2w(splitStr[0].slice(0, 1) + '0') + ' ';
123           tobeParsed = splitStr[0].slice(-1);
124         }
125       } else {
126         tobeParsed += splitStr[0];
127       }
128       for (var keys in ordinals) {
129         if (ordinals[keys][0] === Number(tobeParsed)) {
130           parsedStr += keys + ' ';
131           tobeParsed = '';
132         }
133       }
134 
135       if (splitStr[1].slice(0, 1) === '0') {
136         splitStr[1] = splitStr[1].slice(1);
137       }
138       for (var keys in date) {
139         if (keys === splitStr[1]) {
140           parsedStr += date[keys] + ' ';
141         }
142       }
143       parsedStr += n2w(splitStr[2]);
144     }
145     return parsedStr;
146   }
147 }
